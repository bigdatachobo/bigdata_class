import matplotlib.pyplot as plt
scaler = StandardScaler()
scaler.fit(X)
X_scaled = scaler.transform(X)

fig, axes = plt.subplots(5, 2, figsize=(10, 20))
ax = axes.ravel()

eps_list=[i for i in range(1,11)]
for i in range(10):
    dbscan = DBSCAN(eps=eps_list[i]/10, min_samples=8) #  -> 0.5 
    clusters = dbscan.fit_predict(X_scaled)
    ax[i].scatter(X_scaled[:, 0], 
                  X_scaled[:, 1], 
                  c=clusters, 
                  cmap='viridis', 
                  s=60, 
                  edgecolors='black')
    ax[i].set_title(f'eps : {eps_list[i]/10}')
    ax[i].set_xlabel(" 0")
    ax[i].set_ylabel(" 1")
    print('eps={0}   : {1}'.format((i+1)/10,len(list(filter(lambda x:x==-1,clusters)))))
fig.tight_layout()
