--1번 문제
CREATE  TABLE   customers(
    CNO     NUMBER(5)       PRIMARY KEY,
    CNAME   VARCHAR2(10)    NOT NULL,
    ADDRESS VARCHAR2(50)    NOT NULL,
    EMAIL   VARCHAR2(20)    NOT NULL,
    PHONE   VARCHAR2(20)    NOT NULL
    );
    
CREATE  TABLE   orders(
    ORDERNO     NUMBER(10)      PRIMARY KEY,
    ORDERDATE   DATE            DEFAULT(SYSDATE)     NOT NULL,
    ADDRESS     VARCHAR2(50)    NOT NULL,
    PHONE       VARCHAR2(20)    NOT NULL,
    STATUS      VARCHAR2(20)    CHECK(STATUS IN ('결제완료', '배송중', '배송완료')) NOT NULL,
    CNO         NUMBER(5)       
    );
    
ALTER   TABLE   ORDERS
ADD FOREIGN KEY (CNO) REFERENCES CUSTOMERS(CNO);

CREATE  TABLE   PRODUCTS(
    PNO         NUMBER(5)       PRIMARY KEY,
    PNAME       VARCHAR(20)     NOT NULL,
    COST        NUMBER(8)       NOT NULL,
    STOCK       NUMBER(5)       NOT NULL
    );
    
CREATE  TABLE   ORDERDETAIL(
    ORDERNO     NUMBER(10),
    PNO         NUMBER(5),
    QTY         NUMBER(5)   DEFAULT 0,
    COST        NUMBER(8)   DEFAULT 0,
    PRIMARY KEY (ORDERNO, PNO),
    FOREIGN KEY (ORDERNO)   REFERENCES  ORDERS(ORDERNO),
    FOREIGN KEY (PNO)       REFERENCES  PRODUCTS(PNO)
    );

SELECT  O.ORDERNO, O.ORDERDATE, CM.CNAME, O.ADDRESS, O.PHONE, O.STATUS, PR.PNAME, OD.COST, OD.QTY
FROM    ORDERS          O
JOIN    ORDERDETAIL     OD      ON(O.ORDERNO = OD.ORDERNO) 
JOIN    CUSTOMERS       CM      ON(O.CNO = CM.CNO)
JOIN    PRODUCTS        PR      ON(OD.PNO = PR.PNO);

JOIN    ORDERDETAIL     OD      ON(O.ORDERNO = OD.ORDERNO)
